[docs]def blit(im1,im2,pos=[0,0],mask=None):
    
    # xp1,yp1,xp2,yp2 = blit area on im2
    # x1,y1,x2,y2 = area of im1 to blit on im2
    xp,yp = pos
    x1 = max(0,-xp)
    y1 = max(0,-yp)
    h1,w1 =  im1.shape[:2]
    h2,w2 =  im2.shape[:2]
    xp2 = min(w2, xp+w1)
    yp2 = min(h2, yp+h1)
    x2 = min(w1, w2-xp)
    y2 = min(h1, h2-yp)
    xp1 = max(0,xp)
    yp1 = max(0,yp)
       
    if (xp1>=xp2) or (yp1>=yp2):
        return im2
    
    blitted = im1[y1:y2,x1:x2]
    
    new_im2 = +im2
    
    if mask != None:
        mask = mask[y1:y2,x1:x2]
        if len(im1.shape)==3:
            mask = np.dstack(3*[mask])
        blit_region = new_im2[yp1:yp2,xp1:xp2]
        new_im2[yp1:yp2,xp1:xp2] = (1.0*mask*blitted + (1.0-mask)*blit_region)
    else:
        new_im2[yp1:yp2,xp1:xp2] = blitted
    
    return new_im2.astype('int') if (mask!=None) else new_im2

[docs]def apply_to_mask(f,clip, *a, **k):
    """ This decorator will apply the same function f to the mask of
        the clip created with f """
    newclip = f(clip,*a,**k)
    if hasattr(newclip,'mask'):
        mask = newclip.mask
        if mask !=None:
            newclip.mask = f(mask,*a,**k)
    return newclip

        """
        fl : gf,t -> frame at t originating from function gf
        kwargs may vary according to rthe subclass """
        gf = copy(self.get_frame)
        newclip = self.set_get_frame(lambda t: fl(gf,t))
        
        if isinstance(applyto,str):
			applyto = [applyto]
			
        for attr in applyto:
            if hasattr(newclip,attr):
                a = getattr(newclip,attr)
                if a!=None:
                    setattr(newclip,attr,a.fl(fl))
                    
        return newclip

[docs]    def blit_on(self,frame,t):
        
        ct = t - self.start #clip time
        mask = (None if (self.mask is None)
                         else self.mask.get_frame(ct))
        
        # Set position
        pos = list(self.pos(ct))
        img = self.get_frame(ct)
        hf,wf = frame.shape[:2]
        hi,wi = img.shape[:2]
        if self.relative_pos:
            for i,dim in enumerate(wf,hf):
                if not isinstance(pos[i],str):
                    pos[i] = dim*pos[i]
                    
        if isinstance(pos[0],str):
            D = {'left':0,'center':(wf-wi)/2,'right':wf-wi}
            pos[0]=D[pos[0]]
            
        if isinstance(pos[1],str):
            
            D = {'top':0,'center':(hf-hi)/2,'bottom':hf-hi}
            pos[1]=D[pos[1]]
            
        pos = map(int,pos)
        return blit(img,frame,pos,mask=mask)

        """ clip.resize((360,240)) """
        if self.mask == None:
            fl = lambda gf,t: 1.0*imresize((255*gf(t)).astype('uint8'),
                                    newres,interp='cubic')/255

        """ pos can be
            - a tuple (x,y) where x,y are either integers, or x can be
              'left','center','right', and y can be 'top', 'center',
              'bottom'.
            - a a function lambda t: some position

            newclip.mask = self.mask.set_pos(pos)
        return newclip

        """ A VideoClip originating from a movie file """
        VideoClip.__init__(self)
        self.cap = cv2.VideoCapture(source)
        self.source = source
        self.fps = self.cap.get(cv2.CAP_PROP_FPS)
        self.nframes = self.cap.get(cv2.CAP_PROP_FRAME_COUNT)
        self.duration = 1.0*self.nframes/self.fps
            
        
        self.res = self.cap.read()[1].shape[:2]
        self.last_frame_read = 0
        
        if mask:
            self.mask = mask
        else:
            self.mask = ColorClip(self.res,1.0,mask=False) # white
        
        def get_frame(t):
            """ Gets the frame of the video at time t.
                For development: getting an arbitrary frame in the
                video with opencv can be painfully slow if some decoding
                has to be made. This function tries to avoid fectching
                arbitrary frames whenever possible.
            """
            nframe = int(self.fps*t)
            
            if nframe == self.last_frame_read:
                return self.cap.retrieve()[1][:,:,::-1]
            elif  (self.last_frame_read< nframe < (self.last_frame_read + 5)):
                for i in range(nframe - self.last_frame_read-1):
                    self.cap.read()
                    
            else:
                self.cap.set(cv2.CAP_PROP_POS_FRAMES,nframe-1)
            self.last_frame_read = nframe
            return self.cap.read()[1][:,:,::-1]
        
        self.get_frame = get_frame
        
        if sound:
            temp = 'temp.wav'
            try:
                avconv.extract_sound(source,temp)
                self.sound = SoundClip(temp)
                os.remove(temp)
            except:
                print "No sound detected in %s."%source
